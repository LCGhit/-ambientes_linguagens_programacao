def printStars():
    someInt = int(input('Input a number'))
    null = 0
    for x in range(1,someInt+1, 2):
        null = int((someInt-x)/2)
        print(' '*null + '*'*x + ' '*null)

      result = 0
      for x in list:
          if(x>result):
              result = x
      print(result)

  higherNum([34,60,101,200])
